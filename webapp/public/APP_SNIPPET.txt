// Zeigt die Übungs-Vorschau mit Video an (zuverlässiger auf Mobile)
function showExercisePreview(exercise) {
    if (!exercise) return;
    
    const previewContainer = document.getElementById('exercise-preview');
    const videoEl = document.getElementById('preview-video');
    const imgFallback = document.getElementById('preview-gif-fallback');
    
    if (!previewContainer || !videoEl) return;
    
    // Pfad zum GIF holen
    const gifPath = getExerciseGif(exercise.name);
    if (!gifPath) {
        previewContainer.style.display = 'none';
        return;
    }
    
    // Cache-Busting
    const freshPath = gifPath + '&t=' + Date.now();
    
    // Chrome Mobile: Video statt IMG (zuverlässiger)
    videoEl.src = freshPath;
    videoEl.style.display = 'block';
    videoEl.muted = true;
    videoEl.playsInline = true;
    videoEl.loop = true;
    
    // Video abspielen mit Retry
    const playVideo = () => {
        videoEl.play().catch(err => {
            console.log('Video Play Fehler, Fallback:', err);
            videoEl.style.display = 'none';
            imgFallback.src = freshPath;
            imgFallback.style.display = 'block';
        });
    };
    
    videoEl.onloadeddata = playVideo;
    videoEl.onerror = () => {
        videoEl.style.display = 'none';
        imgFallback.src = freshPath;
        imgFallback.style.display = 'block';
    };
    
    previewContainer.style.display = 'block';
}

// Video manuell neu laden
function reloadGif() {
    const videoEl = document.getElementById('preview-video');
    if (videoEl && videoEl.src) {
        videoEl.currentTime = 0;
        videoEl.play();
    }
}

function closeExerciseSelector() {
    const modal = document.getElementById('exercise-selector-modal');
    if (modal) modal.remove();
}

// Update select dropdowns nach Kategorie gruppiert (Judo + Muskelgruppen)
function updateExerciseSelects() {
    const workoutSelect = document.getElementById('workout-exercise');
    const statsSelect = document.getElementById('stats-exercise');
    const singleSelect = document.getElementById('single-exercise-select');
    
    // Judo-Übungen erkennen (nach Namen)
    const judoKeywords = ['uchi', 'nage', 'randori', 'kata', 'judo', 'ne-waza', 'grip', 'turn-uchi', 'sprungs', 'wurf'];
    const judoExercises = [];
    const otherExercises = [];
    
    exercises.forEach(e => {
        const lowerName = e.name.toLowerCase();
        const isJudo = judoKeywords.some(keyword => lowerName.includes(keyword));
        if (isJudo) {
            judoExercises.push(e);
        } else {
            otherExercises.push(e);
        }
    });
    
    // Optionen erstellen
    const createGroupedOptions = (select) => {
        if (!select) return;
        select.innerHTML = '<option value="">Wählen...</option>';
        
        // Judo-Gruppe
        if (judoExercises.length > 0) {
            const judoGroup = document.createElement('optgroup');
            judoGroup.label = '🥷 Judo';
            judoExercises.sort((a,b) => a.name.localeCompare(b.name));
            judoExercises.forEach(e => {
                const opt = document.createElement('option');
                opt.value = e.id;
                opt.textContent = e.name;
                opt.dataset.muscle = e.muscle_group;
                judoGroup.appendChild(opt);
            });
            select.appendChild(judoGroup);
        }
        
        // Muskelgruppen
        const groups = {};
        otherExercises.forEach(e => {
            if (!groups[e.muscle_group]) groups[e.muscle_group] = [];
            groups[e.muscle_group].push(e);
        });
        
        Object.keys(groups).sort().forEach(muscle => {
            const group = document.createElement('optgroup');
            const icon = getMuscleIcon(muscle);
            group.label = `${icon} ${muscle}`;
            
            groups[muscle].sort((a,b) => a.name.localeCompare(b.name));
            groups[muscle].forEach(e => {
                const opt = document.createElement('option');
                opt.value = e.id;
                opt.textContent = e.name;
                opt.dataset.muscle = e.muscle_group;
                group.appendChild(opt);
            });
            select.appendChild(group);
        });
    };
    
    createGroupedOptions(workoutSelect);
    createGroupedOptions(statsSelect);
    createGroupedOptions(singleSelect);
}

// Icon für Muskelgruppe
function getMuscleIcon(muscle) {
    const icons = {
        'Brust': '💪',
        'Rücken': '🏋️',
        'Beine': '🦵',
        'Schultern': '💪',
        'Arme': '💪',
        'Bauch': '🧘',
        'Mobilität': '🧘',
        'Dehnen': '🧘'
    };
    return icons[muscle] || '💪';
}

function addExercise(event) {
    event.preventDefault();
    const name = document.getElementById('exercise-name').value;
    const muscle_group = document.getElementById('exercise-muscle').value;
    const exercise_type = document.getElementById('exercise-type').value || 'strength';
    
    apiFetch('/api/exercises', {
        method: 'POST',
        body: JSON.stringify({ name, muscle_group, exercise_type })
    })
        .then(res => res.json())
        .then(data => {
            exercises.push(data);
            updateExerciseSelects();
            document.getElementById('exercise-form').reset();
            loadExercises();
            alert('Übung hinzugefügt!');
        });
}

function addWorkout(event) {
    event.preventDefault();
    const exercise_id = document.getElementById('workout-exercise').value;
    const weight = document.getElementById('workout-weight').value;
    const sets = document.getElementById('workout-sets').value;
    const reps = document.getElementById('workout-reps').value;
    const duration = document.getElementById('workout-duration').value;
    const feeling = parseInt(document.getElementById('workout-feeling').value);
    const rest = parseInt(document.getElementById('workout-rest').value);
    const date = document.getElementById('workout-date').value;
    
    // Ermittelte exercise_type
    const exercise = exercises.find(e => e.id == exercise_id);
    const exercise_type = exercise?.exercise_type || 'strength';
    
    const body = { 
        exercise_id, 
        weight, 
        sets, 
        reps,
        duration_seconds: exercise_type === 'time' && duration ? parseInt(duration) : null,
        feeling: isNaN(feeling) ? 5 : feeling,
        rest_seconds: isNaN(rest) ? 60 : rest,
        date 
    };
    
    apiFetch('/api/workouts', {
        method: 'POST',
        body: JSON.stringify(body)
    })
        .then(res => res.json())
        .then(data => {
            loadWorkouts();
            loadStats();
            // Formular zurücksetzen aber Datum behalten
            document.getElementById('workout-weight').value = '';
            document.getElementById('workout-sets').value = '';
            document.getElementById('workout-reps').value = '';
            document.getElementById('workout-duration').value = '';
        });
}

function loadExercises() {
    apiFetch('/api/exercises')
        .then(res => res.json())
        .then(data => {
            exercises = data;
            updateExerciseSelects();
        });
}

function deleteExercise(id) {
    if (!confirm('Wirklich löschen?')) return;
    
    apiFetch(`/api/exercises/${id}`, { method: 'DELETE' })
        .then(() => {
            exercises = exercises.filter(e => e.id !== id);
            updateExerciseSelects();
        });
}

function loadWorkouts() {
    apiFetch('/api/workouts')
        .then(res => res.json())
        .then(data => {
            workouts = data;
            displayWorkouts(workouts);
            // Übersicht aktualisieren
            updateProgressSummary(workouts);
            // Charts aktualisieren
            updateAllCharts();
        });
}

function displayWorkouts(workouts) {
    const list = document.getElementById('workouts-list');
    if (!list) return;
    list.innerHTML = '';
    
    // Nach Datum sortieren (neueste zuerst)
    const sorted = [...workouts].sort((a, b) => new Date(b.date) - new Date(a.date));
    
    // Nach Datum gruppieren
    const grouped = {};
    sorted.forEach(w => {
        if (!grouped[w.date]) grouped[w.date] = [];
        grouped[w.date].push(w);
    });
    
    Object.keys(grouped).sort().reverse().forEach(date => {
        const dateGroup = document.createElement('div');
        dateGroup.style.marginBottom = '15px';
        
        const dateHeader = document.createElement('div');
        dateHeader.style.cssText = 'color: #00d4ff; font-weight: bold; margin-bottom: 8px; font-size: 0.9rem;';
        const formattedDate = new Date(date).toLocaleDateString('de-DE', { weekday: 'short', day: 'numeric', month: 'short' });
        dateHeader.textContent = `📅 ${formattedDate}`;
        dateGroup.appendChild(dateHeader);
        
        grouped[date].forEach(w => {
            const item = document.createElement('div');
            item.className = 'workout-item';
            
            // Bestimme ob Zeit-basierte Übung
            const isTimeExercise = w.exercise_type === 'time' || 
                                    (w.duration_seconds && w.duration_seconds > 0 && (!w.weight || w.weight == 0));
            
            let content = '';
            if (isTimeExercise) {
                // Zeit-basierte Anzeige (ADIM, Plank, etc.)
                const duration = w.duration_seconds ? Math.floor(w.duration_seconds / 60) + ' min' : w.reps + ' sek';
                content = `
                    <div style="display: flex; justify-content: space-between; align-items: center; flex-wrap: wrap; gap: 10px;">
                        <div style="display: flex; align-items: center; gap: 10px;">
                            <strong style="color: #fff;">${w.exercise_name || 'Unbekannt'}</strong>
                            <span style="color: #00d4ff; background: rgba(0,212,255,0.2); padding: 2px 8px; border-radius: 4px; font-size: 0.85rem;">
                                ⏰ ${duration}
                            </span>
                            <span style="color: #888; font-size: 0.85rem;">${w.muscle_group || ''}</span>
                        </div>
                        <div style="display: flex; gap: 5px;">
                            ${feelingEmoji(w.feeling)} 
                            <span style="color: #888; font-size: 0.8rem;">${w.feeling}/10</span>
                            <button class="btn-secondary" onclick="editWorkout(${w.id})" style="padding: 4px 8px; font-size: 0.8rem;">✏️</button>
                            <button class="btn-secondary" onclick="deleteWorkout(${w.id})" style="padding: 4px 8px; font-size: 0.8rem; color: #ff4444;">🗑️</button>
                        </div>
                    </div>
                `;
            } else {
                // Kraft-basierte Anzeige
                content = `
                    <div style="display: flex; justify-content: space-between; align-items: center; flex-wrap: wrap; gap: 10px;">
                        <div style="display: flex; align-items: center; gap: 10px;">
                            <strong style="color: #fff;">${w.exercise_name || 'Unbekannt'}</strong>
                            <span style="color: #00d4ff; background: rgba(0,212,255,0.2); padding: 2px 8px; border-radius: 4px; font-size: 0.85rem;">
                                ${w.weight}kg × ${w.sets} × ${w.reps}
                            </span>
                            <span style="color: #888; font-size: 0.85rem;">${w.muscle_group || ''}</span>
                        </div>
                        <div style="display: flex; gap: 5px;">
                            ${feelingEmoji(w.feeling)} 
                            <span style="color: #888; font-size: 0.8rem;">${w.feeling}/10</span>
                            <button class="btn-secondary" onclick="editWorkout(${w.id})" style="padding: 4px 8px; font-size: 0.8rem;">✏️</button>
                            <button class="btn-secondary" onclick="deleteWorkout(${w.id})" style="padding: 4px 8px; font-size: 0.8rem; color: #ff4444;">🗑️</button>
                        </div>
                    </div>
                `;
            }
            
            item.innerHTML = content;
            dateGroup.appendChild(item);
        });
        
        list.appendChild(dateGroup);
    });
    
    // "Ältere anzeigen"-Button wenn viele Workouts
    if (workouts.length > 20) {
        const loadMore = document.createElement('button');
        loadMore.className = 'btn-secondary';
        loadMore.style.width = '100%';
        loadMore.style.marginTop = '20px';
        loadMore.textContent = `Ältere Workouts anzeigen (${workouts.length} gesamt)`;
        list.appendChild(loadMore);
    }
}

function editWorkout(id) {
    const workout = workouts.find(w => w.id === id);
    if (!workout) return;
    
    // Neues Modal für Bearbeitung
    const modal = document.createElement('div');
    modal.style.cssText = 'position: fixed; top: 0; left: 0; right: 0; bottom: 0; background: rgba(0,0,0,0.8); z-index: 1000; display: flex; align-items: center; justify-content: center; padding: 20px;';
    
    modal.innerHTML = `
        <div style="background: #1a1a2e; padding: 20px; border-radius: 12px; max-width: 400px; width: 100%;">
            <h3 style="margin-bottom: 20px;">✏️ Workout bearbeiten</h3>
            <form id="edit-workout-form">
                <div class="form-group">
                    <label>Gewicht (kg)</label>
                    <input type="number" id="edit-weight" value="${workout.weight || ''}" step="0.5" style="background: rgba(0,0,0,0.3); border: 1px solid rgba(255,255,255,0.1); color: #fff; padding: 10px; border-radius: 8px; width: 100%;">
                </div>
                <div class="form-row">
                    <div class="form-group" style="flex: 1;">
                        <label>Sätze</label>
                        <input type="number" id="edit-sets" value="${workout.sets || ''}" style="background: rgba(0,0,0,0.3); border: 1px solid rgba(255,255,255,0.1); color: #fff; padding: 10px; border-radius: 8px; width: 100%;">
                    </div>
                    <div class="form-group" style="flex: 1;">
                        <label>Wdh</label>
                        <input type="number" id="edit-reps" value="${workout.reps || ''}" style="background: rgba(0,0,0,0.3); border: 1px solid rgba(255,255,255,0.1); color: #fff; padding: 10px; border-radius: 8px; width: 100%;">
                    </div>
                </div>
                <div class="form-group">
                    <label>Feeling (1-10)</label>
                    <input type="range" id="edit-feeling" min="1" max="10" value="${workout.feeling || 5}" style="width: 100%;" oninput="document.getElementById('edit-feeling-display').textContent = this.value">
                    <span id="edit-feeling-display" style="color: #00d4ff; font-size: 1.2rem;">${workout.feeling || 5}</span>/10
                </div>
                <div style="display: flex; gap: 10px; margin-top: 20px;">
                    <button type="button" class="btn-secondary" onclick="this.closest('.modal').remove()" style="flex: 1;">Abbrechen</button>
                    <button type="submit" class="btn-primary" style="flex: 1;">Ändern</button>
                </div>
            </form>
        </div>
    `;
    
    modal.classList.add('modal');
    document.body.appendChild(modal);
    
    // Form-Handler
    modal.querySelector('form').addEventListener('submit', (e) => {
        e.preventDefault();
        
        const updates = {
            weight: parseFloat(document.getElementById('edit-weight').value) || workout.weight,
            sets: parseInt(document.getElementById('edit-sets').value) || workout.sets,
            reps: parseInt(document.getElementById('edit-reps').value) || workout.reps,
            feeling: parseInt(document.getElementById('edit-feeling').value) || workout.feeling
        };
        
        apiFetch(`/api/workouts/${id}`, {
            method: 'PUT',
            body: JSON.stringify({
                ...workout,
                ...updates,
                exercise_id: workout.exercise_id,
                date: workout.date
            })
        })
            .then(res => res.json())
            .then(() => {
                modal.remove();
                loadWorkouts();
            });
    });
}

function deleteWorkout(id) {
    if (!confirm('Workout löschen?')) return;
    
    apiFetch(`/api/workouts/${id}`, { method: 'DELETE' })
        .then(() => {
            loadWorkouts();
            loadStats();
        });
}

function feelingEmoji(level) {
    if (level >= 9) return '🔥';
    if (level >= 7) return '💪';
    if (level >= 5) return '🙂';
    if (level >= 3) return '😐';
    return '😩';
}

function loadStats() {
    apiFetch('/api/stats')
        .then(res => res.json())
        .then(data => {
            document.getElementById('total-volume').textContent = data.total_volume.toLocaleString();
            document.getElementById('total-workouts').textContent = data.total_workouts;
            document.getElementById('weekly-volume').textContent = data.weekly_volume.toLocaleString();
            document.getElementById('total-exercises').textContent = data.total_exercises;
        });
}
